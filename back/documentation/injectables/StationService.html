import { HttpService, Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { SwitchState } from './models/switch-state.interface';
import { Measurement } from './models/mesurement.entity';

@Injectable()
export class StationService {

    private rootUrl: string;
    private ressourcesUrl: string;
    private devicesUrl: string;
    private deviceId: string;

    constructor(
        private readonly httpService: HttpService,
        private readonly configService: ConfigService) {
        this.rootUrl = this.configService.get('THINGER_IO_API_URL');
        this.ressourcesUrl = this.configService.get('THINGER_IO_API_RESSOURCES_PATH');
        this.devicesUrl = this.configService.get('THINGER_IO_API_DEVICES_PATH');
        this.deviceId = this.configService.get('THINGER_IO_API_DEVICE_ID');
    }

    private async getDeviceState(ressource: string): Promise<SwitchState> {       
        const response = await this.httpService.get(`${this.rootUrl}${this.ressourcesUrl}${ressource}-state`).toPromise();
        return response.data;
    }

    private async switchDevice(path: string, switchState: SwitchState): Promise<any> {
        const response = await this.httpService.post(`${this.rootUrl}${this.ressourcesUrl}${path}`, { in: switchState.state === 'ON' }).toPromise();
        return response.data;
    }

    public async getDevices(): Promise<any> {
        const response = await this.httpService.get(`${this.rootUrl}${this.devicesUrl}`).toPromise();
        return response.data; 
    }

    public async getDevice(): Promise<any> {
        const response = await this.httpService.get(`${this.rootUrl}${this.devicesUrl}/${this.deviceId}`).toPromise();
        return response.data; 
    }

    public switchRessource(ressource: string, switchOnOFF: SwitchState): Promise<void> {
        return this.switchDevice(ressource, switchOnOFF);
    }

    public async getState(resource: string): Promise<SwitchState> {
        return this.getDeviceState(`${resource}`);
    }

    public async getMeteo(): Promise<Measurement> {
        const response = await this.httpService.get(`${this.rootUrl}${this.ressourcesUrl}meteo`).toPromise();
        return response.data;
    }

    public async reset(): Promise<void> {
        await this.httpService.post(`${this.rootUrl}${this.ressourcesUrl}reseting`).toPromise();
    }
}
