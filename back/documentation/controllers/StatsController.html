import { Controller, Get } from '@nestjs/common';
import { ApiHeader, ApiTags } from '@nestjs/swagger';
import { MeteoStats } from './models/meteo-stats.entity';
import { StatsService } from './stats.service';
@ApiHeader({
  name: 'Authorization',
  description: 'Bearer xxxxxxxxxxxxxxxxxxxxxxxx',
})
@ApiTags('Stats')
@Controller('station/stats')
export class StatsController {

  constructor(
    private readonly statsService: StatsService,   
  ){}
    
  @Get('daily')
  getStatsDaily(): Promise<MeteoStats>{
    return this.statsService.getMeteoStatsDaily();
  }

  @Get('h24')
  getStatsH24(): Promise<MeteoStats>{
    return this.statsService.getMeteoStatsH24();
  }
}
